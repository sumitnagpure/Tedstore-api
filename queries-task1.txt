# Display Category with max products
>> Category.objects.annotate(num_products=Count('product')).order_by('-num_products').first()

# Find products within a specific price range
>> Product.objects.filter(base_price__gte=800, base_price__lte=1100)

>> Product.objects.filter(category__name="Ac", subcategory_a="1")

# Find products with atleast one image
>> Product.objects.filter(productimages__isnull=False).distinct()

# Find details and average price for each category
>> from django.db.models import Avg
>> Product.objects.all().aggregate(Average_Price=Avg('base_price'))

# Find products with the highest discounted price within each category
>> Product.objects.values('category__name').annotate(Highest_Discounted_Price=Max('discounted_price')).order_by('category__name')
